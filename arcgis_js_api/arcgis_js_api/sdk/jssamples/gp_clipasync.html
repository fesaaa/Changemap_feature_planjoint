Description

This sample shows how to work with an asynchronous geoprocessing service. The Extract Data Task lets end users specify an area of interest then download a zip file that contains data for that area from one or more layers. The Extract Data Task requires several input parameters including the area of interest, the layers to include in the zip file and the download format. This sample constructs a simple user interface to gather these inputs from the user.

The Extract Data Task is an asynchronous geoprocessing service so we'll use the Geoprocessor's submitJob method to execute the task. The submitJob method takes the input parameters and a status callback and complete callback.

gp.submitJob(params, completeCallback , statusCallback);

The status callback is called each time the geoprocessor checks the status of the function. The jobInfo parameter contains the job id, the current status and GPMessages (if enabled).

function statusCallback(jobInfo) {
var status = jobInfo.jobStatus;
}

The complete callback is called when the task completes, in this example, we check to see if the job was successful and then execute a function that will download the newly created zip file.

function completeCallback(jobInfo){
if(jobInfo.jobStatus !== "esriJobFailed"){
gp.getResultData(jobInfo.jobId,"Output_Zip_File", downloadFile);
}
}

    <script>
      dojo.require("dijit.layout.BorderContainer");
      dojo.require("dojox.layout.FloatingPane");
      dojo.require("dijit.layout.ContentPane");
      dojo.require("dijit.form.CheckBox");
      dojo.require("dijit.form.ComboBox");
      dojo.require("esri.map");

      var gp, map;
      var loading;
      var selectionToolbar;

      function init() {
        esri.config.defaults.io.proxyUrl = "/proxy";
        loading = dojo.byId("loadingImg");

        map = new esri.Map("map", {
          basemap: "streets",
          center: [-77.026, 38.905],
          zoom: 14
        });

        dojo.connect(map, 'onLoad', function(map) {
          initSelectionToolbar();
        });
        
        var homelandSecurity = new esri.layers.ArcGISDynamicMapServiceLayer("http://sampleserver4.arcgisonline.com/ArcGIS/rest/services/HomelandSecurity/Incident_Data_Extraction/MapServer");
        map.addLayer(homelandSecurity);
        
        gp = new esri.tasks.Geoprocessor("http://sampleserver4.arcgisonline.com/ArcGIS/rest/services/HomelandSecurity/Incident_Data_Extraction/GPServer/Extract%20Data%20Task");
        gp.setOutSpatialReference({wkid:102100});
      }
      
      function initSelectionToolbar(myMap){
        selectionToolbar = new esri.toolbars.Draw(map);
        dojo.connect(selectionToolbar, "onDrawEnd", function(geometry) {
          selectionToolbar.deactivate();
          var symbol = new esri.symbol.SimpleFillSymbol(esri.symbol.SimpleFillSymbol.STYLE_SOLID, new esri.symbol.SimpleLineSymbol(esri.symbol.SimpleLineSymbol.STYLE_DASHDOT, new dojo.Color([255,0,0]), 2), new dojo.Color([255,255,0,0.25]));
          var graphic = new esri.Graphic(geometry, symbol);
          map.graphics.add(graphic);

        });
      }

      function activateTool(type){
        selectionToolbar.activate();
      }
      
      function extractData(){
        //get clip layers
        var clipLayers = [];
        if(dijit.byId('layer1').checked){clipLayers.push('Incident Points');}
        if(dijit.byId('layer2').checked){clipLayers.push('Incident Lines');}
        if(dijit.byId('layer3').checked){clipLayers.push('Incident Areas');}
        if(clipLayers.length === 0 || map.graphics.graphics.length === 0){
          alert('Select layers to extract and area of interest');
          return;
        }
        var features =[];
        features.push(map.graphics.graphics[0]);
        var featureSet = new esri.tasks.FeatureSet();
        featureSet.features = features;

        var params = {"Layers_to_Clip":clipLayers,
          "Area_of_Interest": featureSet,
          "Feature_Format": dijit.byId('formatBox').value
        }
        esri.show(loading);
        gp.submitJob(params, completeCallback , statusCallback,function(error){
          alert(error);
          esri.hide(loading);
        });
      }
      function completeCallback(jobInfo){
        if(jobInfo.jobStatus !== "esriJobFailed"){
          gp.getResultData(jobInfo.jobId,"Output_Zip_File", downloadFile);
        }
      }
      function statusCallback(jobInfo) {
        var status = jobInfo.jobStatus;
        if(status === "esriJobFailed"){
          alert(status);
          esri.hide(loading);
        }
        else if (status === "esriJobSucceeded"){
          esri.hide(loading);
        }
      }
      function downloadFile(outputFile){
        map.graphics.clear();
        var theurl = outputFile.value.url;  
        window.location = theurl;
      }

      //show map on load 
      dojo.ready(init);
    </script>
